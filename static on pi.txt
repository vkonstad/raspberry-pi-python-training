# show connections
nmcli con show 

# -> manual mode
sudo nmcli con mod "MyHomeWiFi" ipv4.method manual

# ip with subnet mask
sudo nmcli con mod "MyHomeWiFi" ipv4.addresses 192.168.1.100/24

# gateway
sudo nmcli con mod "MyHomeWiFi" ipv4.gateway 192.168.1.1

# DNS servers
sudo nmcli con mod "MyHomeWiFi" ipv4.dns "192.168.1.1,8.8.8.8"
